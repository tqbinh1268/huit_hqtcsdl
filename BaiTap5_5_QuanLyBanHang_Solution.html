📌 Yêu cầu đề bài: Cài đặt giao tác "Thêm mặt hàng mới"
🛠️ Giải pháp mã lệnh T-SQL Proc: 
CREATE PROC sp_ThemMathang
    @MaHG CHAR(10), @TenHG NVARCHAR(100), @DVT NVARCHAR(20), 
    @Soluongton INT, @HangSX NVARCHAR(50), @TinhTrang NVARCHAR(50)
AS
BEGIN
    BEGIN TRY
        BEGIN TRAN
            INSERT INTO HANG(MAHG, TENHG, DVT, SOLUONGTON, HANGSX, TINHTRANG)
            VALUES (@MaHG, @TenHG, @DVT, @Soluongton, @HangSX, @TinhTrang);
        COMMIT TRAN
        PRINT N'Thêm mặt hàng mới thành công!'
    END TRY
    BEGIN CATCH
        ROLLBACK TRAN
        PRINT N'Lỗi thêm mặt hàng mới: ' + ERROR_MESSAGE()
    END CATCH
END

📌 Yêu cầu đề bài: Cài đặt giao tác "Lập phiếu nhập"
🛠️ Giải pháp mã lệnh T-SQL Proc (Đảm bảo tính nhất quán giữa PN, CTPN và kho HANG): 
CREATE PROC sp_LapPhieuNhap
    @MaPN CHAR(10), @NgayNhap DATETIME, @MaNCC CHAR(10), @MaNV CHAR(10),
    @MaHG CHAR(10), @Soluong INT, @GiaNhap MONEY
AS
BEGIN
    DECLARE @ThanhTien MONEY = @Soluong * @GiaNhap;
    BEGIN TRY
        BEGIN TRAN
            -- 1. Thêm thông tin vào bảng PhieuNhap nếu chưa tồn tại
            IF NOT EXISTS (SELECT 1 FROM PHIEUNHAP WHERE MAPN = @MaPN)
            BEGIN
                INSERT INTO PHIEUNHAP(MAPN, NGAYNHAP, MANCC, MANV, TIENNHAP)
                VALUES (@MaPN, @NgayNhap, @MaNCC, @MaNV, @ThanhTien);
            END
            ELSE
            BEGIN
                UPDATE PHIEUNHAP SET TIENNHAP = TIENNHAP + @ThanhTien WHERE MAPN = @MaPN;
            END

            -- 2. Thêm vào chi tiết phiếu nhập
            INSERT INTO CHITIETPN(MAPN, MAHG, SOLUONG, GIANHAP, THANHTIEN)
            VALUES (@MaPN, @MaHG, @Soluong, @GiaNhap, @ThanhTien);

            -- 3. Cập nhật tự động tăng số lượng tồn kho của mặt hàng
            UPDATE HANG SET SOLUONGTON = SOLUONGTON + @Soluong WHERE MAHG = @MaHG;
        COMMIT TRAN
        PRINT N'Lập phiếu nhập kho thành công!'
    END TRY
    BEGIN CATCH
        ROLLBACK TRAN
        PRINT N'Lỗi lập phiếu nhập kho: ' + ERROR_MESSAGE()
    END CATCH
END

✦ Đối với nhân viên bán hàng
📌 Yêu cầu đề bài: Cài đặt giao tác "Tra cứu thông tin mặt hàng"
🛠️ Giải pháp mã lệnh T-SQL Proc (Truy vấn kèm giá mới nhất): 
CREATE PROC sp_TraCuuMatHang
    @MaHG CHAR(10)
AS
BEGIN
    SELECT H.MAHG, H.TENHG, H.DVT, H.SOLUONGTON, DG.GIA AS GiaHienTai
    FROM HANG H 
    LEFT JOIN DONGIA DG ON H.MAHG = DG.MAHG
    WHERE H.MAHG = @MaHG
      AND DG.NGAYCN = (SELECT MAX(NGAYCN) FROM DONGIA WHERE MAHG = @MaHG);
END
📌 Yêu cầu đề bài: Cài đặt giao tác "Bán hàng"
🛠️ Giải pháp mã lệnh T-SQL Proc (Kiểm tra tồn, lập HD, CTHD và trừ kho đồng bộ): 
CREATE PROC sp_BanHang
    @MaHD CHAR(10), @NgayDat DATETIME, @MaKH CHAR(10), @MaNV CHAR(10),
    @MaHG CHAR(10), @SoluongBan INT, @GiaBan MONEY
AS
BEGIN
    DECLARE @ThanhTien MONEY = @SoluongBan * @GiaBan;
    DECLARE @TonKho INT;
    BEGIN TRY
        BEGIN TRAN
            -- Kiểm tra số lượng tồn kho và khóa giữ dòng dữ liệu tránh tranh chấp
            SELECT @TonKho = SOLUONGTON FROM HANG WITH (UPDLOCK) WHERE MAHG = @MaHG;
            
            IF @TonKho < @SoluongBan
            BEGIN
                RAISERROR(N'Số lượng hàng tồn kho không đủ để đáp ứng đơn hàng!', 16, 1);
            END

            -- Thêm thông tin vào bảng HoaDon nếu chưa tồn tại
            IF NOT EXISTS (SELECT 1 FROM HOADON WHERE MAHD = @MaHD)
            BEGIN
                INSERT INTO HOADON(MAHD, NGAYDAT, MAKH, MANV, TIENBAN)
                VALUES (@MaHD, @NgayDat, @MaKH, @MaNV, @ThanhTien);
            END
            ELSE
            BEGIN
                UPDATE HOADON SET TIENBAN = TIENBAN + @ThanhTien WHERE MAHD = @MaHD;
            END

            -- Thêm vào chi tiết hóa đơn
            INSERT INTO CHITIETHD(MAHD, MAHG, SOLUONG, GIABAN, THANHTIEN)
            VALUES (@MaHD, @MaHG, @SoluongBan, @GiaBan, @ThanhTien);

            -- Tự động trừ bớt số lượng tồn kho của mặt hàng
            UPDATE HANG SET SOLUONGTON = SOLUONGTON - @SoluongBan WHERE MAHG = @MaHG;
        COMMIT TRAN
        PRINT N'Thực hiện giao dịch bán hàng thành công!'
    END TRY
    BEGIN CATCH
        ROLLBACK TRAN
        PRINT N'Lỗi giao dịch bán hàng: ' + ERROR_MESSAGE()
    END CATCH
END
✦ Đối với quản lý
📌 Yêu cầu đề bài: Cài đặt giao tác "Thống kê doanh thu bán hàng"
🛠️ Giải pháp mã lệnh T-SQL Proc (Thống kê theo từng tháng trong năm): 
CREATE PROC sp_ThongKeDoanhThuTheoThang
    @Nam INT
AS
BEGIN
    SELECT MONTH(NGAYDAT) AS Thang, SUM(TIENBAN) AS TongDoanhThu
    FROM HOADON
    WHERE YEAR(NGAYDAT) = @Nam
    GROUP BY MONTH(NGAYDAT);
END
2) Giải quyết các tình huống tranh chấp đồng thời theo yêu cầu của đề bài
Tình huống xung đột 1 READ COMMITTED
Đề bài yêu cầu: Nhân viên bán hàng tra cứu giá của một mặt hàng trong lúc nhân viên kho đang cập nhật giá mới cho mặt hàng đó.
⚠️ Nguy cơ / Rủi ro tranh chấp Xảy ra hiện tượng Đọc dữ liệu rác (Dirty Read). Nhân viên bán hàng có thể đọc phải một mức giá ảo đang sửa đổi dở dang của nhân viên kho nhưng chưa được commit. Nếu nhân viên kho rollback hủy lệnh cập nhật giá, thông tin giá bán áp cho khách hàng sẽ bị sai lệch hoàn toàn.
🛡️ Giải pháp xử lý trong SQL Server Cấu hình mức cô lập giao tác tra cứu giá của nhân viên bán hàng về mức READ COMMITTED (đây cũng là mức cô lập mặc định). Giao tác đọc sẽ bị chặn lại cho đến khi nhân viên kho hoàn tất việc cập nhật và gọi lệnh COMMIT.
Tình huống xung đột 2 UPDLOCK / XLOCK
Đề bài yêu cầu: Hai giao dịch bán hàng cùng tranh chấp số lượng tồn của một mặt hàng nào đó tại một thời điểm.
⚠️ Nguy cơ / Rủi ro tranh chấp Xảy ra lỗi nghiêm trọng **Mất dữ liệu cập nhật (Lost Update)**. Cả hai nhân viên bán hàng cùng đọc ra số lượng tồn kho cũ (ví dụ: tồn = 1). Cả hai giao dịch đều nghĩ rằng còn đủ hàng nên tiến hành tạo hóa đơn bán và trừ kho. Giao tác hoàn tất sau ghi đè lên giao tác trước khiến số lượng tồn thực tế bị âm (tồn kho sai lệch).
🛡️ Giải pháp xử lý trong SQL Server Sử dụng Locking hint WITH (UPDLOCK) hoặc WITH (XLOCK) ngay khi câu lệnh đọc check tồn kho trong thủ tục bán hàng thực thi. Khóa này sẽ chặn đứng giao dịch bán hàng thứ hai, bắt buộc phải đợi giao dịch thứ nhất kết thúc trừ kho rồi mới được vào đọc số lượng tồn mới.
Tình huống xung đột 3 SERIALIZABLE
Đề bài yêu cầu: Nhân viên kho thống kê danh sách các mặt hàng gần hết (có số lượng tồn nhỏ hơn con số cụ thể nào đó) trong khi một giao tác bán hàng đang xảy ra.
⚠️ Nguy cơ / Rủi ro tranh chấp Xảy ra lỗi **Không thể đọc lại dữ liệu (Unrepeatable Read)** hoặc **Bóng ma dữ liệu (Phantom Read)**. Trong khi nhân viên kho đang duyệt danh sách dở dang để làm báo cáo, giao tác bán hàng trừ bớt kho của mặt hàng khác khiến mặt hàng đó bất ngờ tụt xuống dưới mức hạn định, làm cho danh sách thống kê ở các trang sau bị thay đổi, báo cáo không đồng nhất dữ liệu.
🛡️ Giải pháp xử lý trong SQL Server Nâng cấp cấp độ cô lập của giao tác thống kê kho của nhân viên kho lên mức tối đa SET TRANSACTION ISOLATION LEVEL SERIALIZABLE;. Hệ quản trị sẽ giữ chặt khóa chia sẻ phạm vi (Range Lock) trên bảng HANG, cố định toàn bộ dữ liệu kho cho tới khi nhân viên kho kết xuất xong báo cáo.
Tình huống xung đột 4 READ COMMITTED
Đề bài yêu cầu: Nhân viên bán hàng tra cứu số lượng tồn kho của mặt hàng trong lúc nhân viên kho đang nhập thêm mặt hàng đó vào kho.
⚠️ Nguy cơ / Rủi ro tranh chấp Xảy ra lỗi **Đọc dữ liệu bẩn (Dirty Read)**. Nhân viên bán hàng nhìn thấy số lượng hàng tồn kho tăng vọt (do nhân viên kho đang làm phiếu nhập tạm thời chưa kiểm đếm xong hoặc chưa commit). Khi nhân viên bán hàng báo với khách còn hàng, đột nhiên nhân viên kho phát hiện hàng hỏng nên hủy phiếu nhập (ROLLBACK), hệ thống quay về số lượng cũ làm mất uy tín cửa hàng.
🛡️ Giải pháp xử lý trong SQL Server Sử dụng mức cô lập READ COMMITTED cho phiên tra cứu của nhân viên bán hàng. Lệnh tra cứu tồn kho sẽ bị hoãn lại một vài giây cho đến khi nhân viên kho thực hiện lệnh lệnh COMMIT phiếu nhập thành công, số lượng hiển thị cho nhân viên bán hàng sẽ là số liệu thực tế 100%.
Tình huống xung đột 5 SNAPSHOT ISOLATION
Đề bài yêu cầu: Quản lý thống kê doanh thu bán hàng trong lúc các giao tác bán hàng đang xảy ra.
⚠️ Nguy cơ / Rủi ro tranh chấp Xảy ra hiện tượng **Bóng ma dữ liệu (Phantom Read)**. Người quản lý chạy lệnh tính tổng tiền liên tục thu được các con số khác nhau do các nhân viên bán hàng ở quầy liên tục tạo hóa đơn mới (INSERT), làm cho báo cáo doanh thu bị thay đổi liên tục, không thể chốt số liệu kế toán cố định tại một thời điểm.
🛡️ Giải pháp xử lý trong SQL Server Giải pháp tối ưu và chuyên nghiệp nhất trong thực tế là kích hoạt mức cô lập dữ liệu ảnh chụp **`SNAPSHOT ISOLATION`** cho giao tác của quản lý. Người quản lý sẽ tính toán doanh thu dựa trên một bản sao (ảnh chụp) ổn định tại thời điểm bắt đầu gọi lệnh, trong khi đó các nhân viên bán hàng ở quầy vẫn chèn mới hóa đơn bình thường mà không bị khóa bảng gây tắc nghẽn hệ thống.
2) Giải quyết các tình huống tranh chấp (Kèm Code Sửa Đổi)
Tình huống xung đột 1 READ COMMITTED
Đề bài: Nhân viên bán hàng tra cứu giá của một mặt hàng trong lúc nhân viên kho đang cập nhật giá mới cho mặt hàng đó.
⚠️ Nguy cơ (Dirty Read) Nhân viên bán hàng có thể đọc phải một mức giá ảo đang sửa đổi dở dang của nhân viên kho. Nếu nhân viên kho rollback hủy lệnh, thông tin giá bán áp cho khách hàng sẽ bị sai lệch hoàn toàn.
🛡️ Giải pháp & Mã lệnh khắc phục Sửa đổi thủ tục tra cứu, ép giao tác chạy ở mức READ COMMITTED và đưa lệnh SELECT vào khối BEGIN TRAN. Lệnh đọc sẽ tự động bị chặn chờ đến khi kho COMMIT. 
ALTER PROC sp_TraCuuMatHang
    @MaHG CHAR(10)
AS
BEGIN
    -- Thiết lập mức cô lập tránh Dirty Read
    SET TRANSACTION ISOLATION LEVEL READ COMMITTED;
    
    BEGIN TRY
        BEGIN TRAN
            SELECT H.MAHG, H.TENHG, H.DVT, H.SOLUONGTON, DG.GIA AS GiaHienTai
            FROM HANG H 
            LEFT JOIN DONGIA DG ON H.MAHG = DG.MAHG
            WHERE H.MAHG = @MaHG
              AND DG.NGAYCN = (SELECT MAX(NGAYCN) FROM DONGIA WHERE MAHG = @MaHG);
        COMMIT TRAN
    END TRY
    BEGIN CATCH
        ROLLBACK TRAN
        PRINT ERROR_MESSAGE()
    END CATCH
END
Tình huống xung đột 2 UPDLOCK / XLOCK
Đề bài: Hai giao dịch bán hàng cùng tranh chấp số lượng tồn của một mặt hàng nào đó tại một thời điểm.
⚠️ Nguy cơ (Lost Update) Hai nhân viên cùng check thấy tồn kho = 1. Cả hai cùng lập hóa đơn và trừ kho. Giao tác sau ghi đè giao tác trước làm tồn kho thực tế bị âm.
🛡️ Giải pháp & Mã lệnh khắc phục Sửa đổi thủ tục sp_BanHang, bổ sung WITH (UPDLOCK) vào câu lệnh SELECT kiểm tra tồn kho. Khóa này chặn ngay giao dịch thứ 2 ngay từ khâu kiểm tra. 
ALTER PROC sp_BanHang
    @MaHD CHAR(10), @NgayDat DATETIME, @MaKH CHAR(10), @MaNV CHAR(10),
    @MaHG CHAR(10), @SoluongBan INT, @GiaBan MONEY
AS
BEGIN
    DECLARE @TonKho INT;
    BEGIN TRY
        BEGIN TRAN
            -- Áp dụng khóa Cập nhật (UPDLOCK) để độc chiếm dòng dữ liệu
            SELECT @TonKho = SOLUONGTON 
            FROM HANG WITH (UPDLOCK) 
            WHERE MAHG = @MaHG;
            
            IF @TonKho < @SoluongBan
                RAISERROR(N'Không đủ hàng!', 16, 1);

            -- (Các lệnh Insert hóa đơn, chi tiết hóa đơn...)

            UPDATE HANG SET SOLUONGTON = SOLUONGTON - @SoluongBan WHERE MAHG = @MaHG;
        COMMIT TRAN
    END TRY
    BEGIN CATCH
        ROLLBACK TRAN
    END CATCH
END
Tình huống xung đột 3 SERIALIZABLE
Đề bài: Nhân viên kho thống kê danh sách các mặt hàng gần hết trong khi một giao tác bán hàng đang xảy ra.
⚠️ Nguy cơ (Phantom Read) Kho đang quét danh sách, bán hàng chen vào trừ kho khiến một mặt hàng bất ngờ lọt xuống nhóm "gần hết" ở nửa sau của báo cáo, gây sai lệch số liệu dòng.
🛡️ Giải pháp & Mã lệnh khắc phục Sửa đổi thủ tục thống kê của kho, nâng mức cô lập lên SERIALIZABLE để khóa toàn bộ dải dữ liệu đang quét. 
ALTER PROC sp_ThongKeHangGanHet
    @HanMuc INT
AS
BEGIN
    -- Mức cô lập cao nhất, khóa phạm vi (Range-key Lock)
    SET TRANSACTION ISOLATION LEVEL SERIALIZABLE;
    
    BEGIN TRAN
        SELECT MAHG, TENHG, DVT, SOLUONGTON, HANGSX
        FROM HANG
        WHERE SOLUONGTON <= @HanMuc;
    COMMIT TRAN
END

ALTER PROC sp_ThongKeDoanhThuTheoThang
    @Nam INT

    -- Sử dụng cơ chế ảnh chụp phiên bản
    SET TRANSACTION ISOLATION LEVEL SNAPSHOT;
    
    BEGIN TRAN
        SELECT MONTH(NGAYDAT) AS Thang, SUM(TIENBAN) AS TongDoanhThu
        FROM HOADON
        WHERE YEAR(NGAYDAT) = @Nam
        GROUP BY MONTH(NGAYDAT);
    COMMIT TRAN
END
